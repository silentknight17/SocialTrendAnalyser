# Environment Variables Template for Production Deployment
# Copy these variables to your Vercel Environment Variables section

# === REQUIRED API KEYS ===
# Groq AI API Key (for hashtag context analysis and text generation)
GROQ_API_KEY=your_groq_api_key_here

# YouTube Data API v3 Key (for YouTube trending data)
YOUTUBE_API_KEY=your_youtube_api_key_here

# === PRODUCTION CONFIGURATION ===
# Node Environment
NODE_ENV=production

# === API RATE LIMITING ===
# Maximum requests per minute per IP
RATE_LIMIT_MAX=100

# Rate limit window in minutes  
RATE_LIMIT_WINDOW=15

# === CACHING CONFIGURATION ===
# Cache timeout for trending data (in minutes)
CACHE_TIMEOUT=15

# News cache timeout (in minutes)
NEWS_CACHE_TIMEOUT=2

# === CORS CONFIGURATION ===
# Allowed origins for CORS (comma-separated)
ALLOWED_ORIGINS=*
